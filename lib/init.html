  Date.prototype.format ||
  function (arg) {
    return format(arg, this);

# toRound 四舍五入

/**
 * 四舍五入法
 * 解决 toFixed 遇到坑。
 * @param {*} num 保留位数
 */
Number.prototype.toRound =
  Number.prototype.toRound ||
  function (num) {
    return toRound(this, num);

# toMoney 数字转金额

/**
 * 数字转换为金额(小数位数,符号,千位标识,小数点标识)
 * @param {*} n 小数位数
 * @param {*} symbol 符号"￥"
 */
Number.prototype.toMoney =
  Number.prototype.toMoney ||
  function (n, symbol = '￥') {
    return toMoney(this, n, symbol);
  };
